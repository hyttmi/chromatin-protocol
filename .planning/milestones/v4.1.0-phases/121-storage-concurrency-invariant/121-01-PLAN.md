  // db/storage/thread_check.h
  //
  // Runtime assertion that Storage is touched from exactly one thread after
  // the first call. Compiles to a no-op under NDEBUG. Matches the
  // "strand-confined to io_context thread" pattern documented in
  // db/peer/peer_types.h:58.
  //
  // Usage (inside Storage::Impl or Storage itself):
  //   struct Impl {
  //     storage::ThreadOwner thread_owner_;
  //     ...
  //   };
  //   StoreResult Storage::store_blob(...) {
  //     STORAGE_THREAD_CHECK();
  //     ...
  //   }
  //
  // D-07/D-08/D-09: lazy capture on first call, must not leak into public API.

  #pragma once
  #include <atomic>
  #include <cassert>
  #include <thread>

  namespace chromatindb::storage {

  class ThreadOwner {
  public:
    // Called at entry of every public Storage method. Captures the current
    // thread id on first invocation; asserts identity on every subsequent call.
    inline void check(const char* method) noexcept {
      std::thread::id cur = std::this_thread::get_id();
      std::thread::id expected = owner_.load(std::memory_order_relaxed);
      if (expected == std::thread::id{}) {
        // First call: attempt to install current tid as owner.
        std::thread::id desired = cur;
        owner_.compare_exchange_strong(expected, desired,
                                       std::memory_order_acq_rel,
                                       std::memory_order_relaxed);
        return;
      }
      assert(expected == cur &amp;&amp; "Storage accessed from a thread other "
             "than the one that first touched it — invariant violated");
      (void)method;
    }

    // Test-only: reset the owner to allow test cases to exercise first-capture
    // semantics on a fresh ThreadOwner without reconstructing Storage.
    void reset_for_test() noexcept {
      owner_.store(std::thread::id{}, std::memory_order_release);
    }

  private:
    // std::thread::id default-constructed == "no thread" sentinel.
    std::atomic<std::thread::id> owner_{std::thread::id{}};
  };

  }  // namespace chromatindb::storage

  #if defined(NDEBUG)
    #define STORAGE_THREAD_CHECK() ((void)0)
  #else
    #define STORAGE_THREAD_CHECK() (this->impl_->thread_owner_.check(__func__))
  #endif
