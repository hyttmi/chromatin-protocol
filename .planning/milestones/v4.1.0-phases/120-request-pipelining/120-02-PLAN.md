```cpp
// 120-02 / PIPE-01: Pipeline the put fan-out.
// Phase A: fire writes up to Connection::kPipelineDepth; backpressure handled
// inside Connection::send_async (which pumps recv() into pending_replies_ when
// full).
// Phase B: drain one WriteAck at a time via recv() in arrival order (D-08),
// look up which file each ack belongs to via the batch-local rid->index map.
//
// recv() (not recv_for) is intentional: D-08 wants per-item lines in
// completion order, not request order. send_async's backpressure pump may
// have already stashed some acks in Connection::pending_replies_; we drain
// those first before pulling more off the wire. This keeps the single-reader
// invariant (PIPE-02) intact: still exactly one thread calls recv().

uint32_t rid = 1;
std::unordered_map<uint32_t, size_t> rid_to_index;  // batch-local
size_t next_to_send = 0;
size_t completed = 0;

while (completed < files.size()) {
    // Phase A: send while we have inputs left. send_async blocks via its
    // internal recv-pump when in_flight_ == kPipelineDepth, so this loop
    // self-regulates without an explicit depth check here.
    if (next_to_send < files.size()) {
        auto& f = files[next_to_send];

        // === Existing payload construction, lifted verbatim from old loop ===
        auto payload = build_put_payload(f.name, f.data);
        auto envelope_data = envelope::encrypt(payload, recipient_spans);
        auto timestamp = static_cast<uint64_t>(std::time(nullptr));
        auto signing_input = build_signing_input(ns, envelope_data, ttl, timestamp);
        auto signature = id.sign(signing_input);

        BlobData blob{};
        std::memcpy(blob.namespace_id.data(), ns.data(), 32);
        blob.pubkey.assign(id.signing_pubkey().begin(), id.signing_pubkey().end());
        blob.data = std::move(envelope_data);
        blob.ttl = ttl;
        blob.timestamp = timestamp;
        blob.signature = std::move(signature);

        auto flatbuf = encode_blob(blob);
        // === End preserved payload construction ===

        uint32_t this_rid = rid++;
        if (!conn.send_async(MsgType::Data, flatbuf, this_rid)) {
            std::fprintf(stderr, "Error: failed to send %s\n", f.name.c_str());
            ++errors;
            ++completed;
            ++next_to_send;
            continue;
        }
        rid_to_index[this_rid] = next_to_send;
        ++next_to_send;
        // Loop back: fill more slots until in_flight reaches depth, at which
        // point send_async drains internally. We choose to keep sending
        // greedily before draining at the top level — this minimizes the
        // window where the pipeline is partially full.
        if (next_to_send < files.size() && rid_to_index.size() < Connection::kPipelineDepth) {
            continue;
        }
    }

    // Phase B: drain one reply. We use recv() directly (not recv_for) so
    // whichever WriteAck arrives first gets handled first (D-08 completion
    // order). The single-reader invariant holds because we are still the
    // only thread calling recv().
    auto resp = conn.recv();
    if (!resp) {
        // Transport dead — every still-pending request becomes an error.
        for (auto& [r, idx] : rid_to_index) {
            std::fprintf(stderr, "Error: connection lost while waiting for %s\n",
                         files[idx].name.c_str());
            ++errors;
            ++completed;
        }
        rid_to_index.clear();
        break;
    }

    auto it = rid_to_index.find(resp->request_id);
    if (it == rid_to_index.end()) {
        // Stray rid (server bug or orphaned reply from a previous batch).
        // Mirrors the connection.cpp recv_for D-04 stance: log, drop, continue.
        spdlog::debug("cmd::put: discarding reply for unknown rid {} (type {})",
                      resp->request_id, static_cast<unsigned>(resp->type));
        continue;
    }

    size_t file_idx = it->second;
    rid_to_index.erase(it);
    ++completed;
    auto& f = files[file_idx];

    if (resp->type != static_cast<uint8_t>(MsgType::WriteAck) ||
        resp->payload.size() < 41) {
        std::fprintf(stderr, "Error: bad response for %s\n", f.name.c_str());
        ++errors;
        continue;
    }

    auto hash_span = std::span<const uint8_t>(resp->payload.data(), 32);
    auto hash_hex = to_hex(hash_span);
    if (!opts.quiet && !f.name.empty() && files.size() > 1) {
        std::printf("%s  %s\n", hash_hex.c_str(), f.name.c_str());
    } else if (!opts.quiet) {
        std::printf("%s\n", hash_hex.c_str());
    }
}
```

```cpp
// Same shape as cmd::put (Task 1 of this plan). recv() drives arrival-order
// drain (D-08). Per-batch rid_to_index maps whichever ReadResponse arrives
// first back to the requesting hash. Connection::kPipelineDepth governs

        const auto& hash_hex = hash_hexes[next_to_send];

        // === Existing per-hash request build, lifted verbatim ===
        auto hash_bytes = parse_hex(hash_hex);
        if (!hash_bytes || hash_bytes->size() != 32) {
            std::fprintf(stderr, "Error: invalid hex %s\n", hash_hex.c_str());
            ++errors;
            ++completed;
            ++next_to_send;
            continue;
        }
        // (Whatever request-payload encoding cmd::get currently does — typically
        // a 32-byte hash payload sent as MsgType::ReadRequest.)
        auto request_payload = encode_read_request(*hash_bytes);  // or inline span
        // === End preserved request build ===

        uint32_t this_rid = rid++;

            std::fprintf(stderr, "Error: failed to send request for %s\n",
                         hash_hex.c_str());
            ++errors;
            ++completed;
            ++next_to_send;
            continue;

        rid_to_index[this_rid] = next_to_send;
        ++next_to_send;

    // Phase B: drain one reply in arrival order.
    auto resp = conn.recv();

        for (auto& [r, idx] : rid_to_index) {
            std::fprintf(stderr, "Error: connection lost while fetching %s\n",
                         hash_hexes[idx].c_str());

            ++completed;

        rid_to_index.clear();

    auto it = rid_to_index.find(resp->request_id);
    if (it == rid_to_index.end()) {
        spdlog::debug("cmd::get: discarding reply for unknown rid {} (type {})",
                      resp->request_id, static_cast<unsigned>(resp->type));
        continue;

        continue;

        continue;

        ++errors;
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| node → cdb client | The remote chromatindb node is the source of every WriteAck and ReadResponse. Pipelining changes the timing/ordering of replies but not the trust model — each reply is still independently AEAD-decrypted by `Connection::recv()` before reaching cmd::get / cmd::put. |
| user input → cdb argv | The user-supplied list of hashes (`cdb get h1 h2 ...`) and files (`cdb put f1 f2 ...`) feeds the per-batch rid_to_index map size. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-120-08 | Denial of Service (memory growth via batch size) | per-batch rid_to_index in cmd::get / cmd::put | mitigate | The map size is bounded by `Connection::kPipelineDepth` (8), not by the user's batch size. The greedy-fill check `rid_to_index.size() < Connection::kPipelineDepth` guarantees at most 8 in-flight rids regardless of how many hashes/files the user supplies. A user passing 10 million hashes will see a steady ~8-entry map with the rest waiting in `next_to_send`. |
| T-120-09 | Tampering / Spoofing (rid mismatch routing wrong reply to wrong file) | rid_to_index lookup | mitigate | The rid is generated locally as a monotonic counter (`uint32_t rid = 1; rid++;`) — the server cannot pick it. The server merely echoes the rid it received in the request (existing wire-format guarantee, unchanged by this phase). A buggy server returning a wrong rid lands in the "unknown rid" branch (logged at debug, dropped). The reply itself is AEAD-protected, so the server cannot inject a forged ack for a rid the user never sent. |
| T-120-10 | Information disclosure (reply content for blob A printed under filename for blob B) | per-item drain block | mitigate | The drain branch looks up `file_idx = rid_to_index[resp->request_id]` and uses `files[file_idx].name` — so the printed filename always matches the rid the server replied to. Verifiable at the human-verify checkpoint via `diff -r /tmp/seq /tmp/pipe` (byte-identical outputs). |
| T-120-11 | Repudiation (reply order doesn't match request order) | per-item line printing | accept | D-08 explicitly accepts arrival-order delivery. Documented in code (`// D-08: completion order, not request order`) and in cli/README.md (Phase 121 will document this user-facing). Not a security issue. |
| T-120-12 | Denial of service (server withholds replies, send_async hangs forever) | send_async backpressure pump | accept | Same residual risk as T-120-03 in Plan 01. The chromatindb node is trusted to either reply or close the connection. Hardening with a per-read deadline is tracked separately as backlog 999.8. |
| T-120-13 | Tampering (transport dies mid-batch, partial state leaks) | "connection lost while fetching" branch | mitigate | When `recv()` returns nullopt, the new code iterates rid_to_index and reports an explicit error per still-pending request (rather than a single generic "connection died" message). User sees exactly which files succeeded vs. failed. Exit code is non-zero, no silent data loss. |
</threat_model>
