Add a documenting test to `db/tests/net/test_handshake.cpp` that explicitly validates the pubkey binding invariant:

```cpp
TEST_CASE("verify_peer_auth rejects auth with mismatched pubkey", "[handshake][binding]") {
    // Create two different identities
    chromatindb::crypto::Identity initiator_id;
    chromatindb::crypto::Identity responder_id;
    chromatindb::crypto::Identity attacker_id;

    // Set up handshake as normal
    HandshakeInitiator hs_init(initiator_id);
    auto kem_pk_msg = hs_init.start();

    HandshakeResponder hs_resp(responder_id);
    auto [kem_ct_msg, resp_err] = hs_resp.receive_kem_pubkey(kem_pk_msg);
    REQUIRE(resp_err == HandshakeError::Success);

    auto init_err = hs_init.receive_kem_ciphertext(kem_ct_msg);
    REQUIRE(init_err == HandshakeError::Success);

    auto init_keys = hs_init.take_session_keys();
    auto resp_keys = hs_resp.take_session_keys();

    // Attacker signs the session fingerprint with their own key
    auto attacker_sig = attacker_id.sign(resp_keys.session_fingerprint);
    auto attacker_payload = chromatindb::net::encode_auth_payload(
        attacker_id.public_key(), attacker_sig);

    // Responder should reject because attacker's pubkey != initiator's pubkey
    // The decode_auth_payload will now reject because attacker's pubkey size IS 2592,
    // so it passes that check. But verify_peer_auth should reject the pubkey mismatch.
    auto decoded = chromatindb::net::decode_auth_payload(attacker_payload);
    REQUIRE(decoded.has_value());  // Payload format is valid

    // verify_peer_auth checks pubkey matches the one from KEM exchange
    bool accepted = hs_resp.verify_peer_auth(*decoded);
    REQUIRE_FALSE(accepted);  // Must reject mismatched pubkey
}
```
