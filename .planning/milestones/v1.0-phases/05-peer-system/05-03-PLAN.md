   ```cpp
   static constexpr const char* VERSION = "0.1.0-dev";

   // Forward declarations
   int cmd_run(int argc, char* argv[]);
   int cmd_keygen(int argc, char* argv[]);
   int cmd_version();
   void print_usage(const char* prog);
   ```

   **`print_usage`:** Print usage with subcommands: run, keygen, version.

   **`cmd_version`:** Print `"chromatindb {VERSION}"` to stdout, return 0.

   **`cmd_keygen`:**
   - Parse args: `--data-dir <path>` (default "./data"), `--force` flag
   - Check if identity files exist in data-dir: `{data-dir}/node.key` and `{data-dir}/node.pub`
   - If exist AND no `--force`: print error "Identity already exists at {path}. Use --force to overwrite." and return 1
   - If `--force` or files don't exist: call `NodeIdentity::generate()`, then `save_to(data_dir)`
   - Print: "Generated identity at {data-dir}/"
   - Print: "Namespace: {hex of namespace_id}"
   - Return 0

   **`cmd_run`:**
   - Parse args using existing `parse_args()` (handles --config, --data-dir, --log-level)
   - Load config from file if --config provided
   - Init logging: `init_logging(config.log_level)`
   - Print startup banner at info level:
     ```
     spdlog::info("chromatindb {}", VERSION);
     spdlog::info("bind: {}", config.bind_address);
     spdlog::info("data: {}", config.data_dir);
     ```
   - Create/load identity: `auto identity = NodeIdentity::load_or_generate(config.data_dir);`
   - Log identity: `spdlog::info("namespace: {}", hex(identity.namespace_id()));`
   - Log bootstrap peers:
     ```cpp
     for (const auto& peer : config.bootstrap_peers) {
         spdlog::info("bootstrap: {}", peer);
     }
     if (config.bootstrap_peers.empty()) {
         spdlog::info("no bootstrap peers configured (accepting inbound only)");
     }
     ```
   - Create components:
     ```cpp
     storage::Storage storage(config.data_dir);
     engine::BlobEngine engine(storage);
     asio::io_context ioc;
     peer::PeerManager pm(config, identity, engine, storage, ioc);
     ```
   - Start: `pm.start();`
   - Run expiry scanner on a periodic timer (co_spawn a coroutine that runs every 60 seconds):
     ```cpp
     asio::co_spawn(ioc, [&]() -> asio::awaitable<void> {
         while (true) {
             asio::steady_timer timer(ioc);
             timer.expires_after(std::chrono::seconds(60));
             co_await timer.async_wait(asio::as_tuple(asio::use_awaitable));
             auto purged = storage.run_expiry_scan();
             if (purged > 0) {
                 spdlog::info("expiry scan: purged {} blobs", purged);
             }
         }
     }(), asio::detached);
     ```
   - Run io_context: `ioc.run();`
   - Return 0

   **`main`:**
   ```cpp
   int main(int argc, char* argv[]) {
       if (argc < 2) {
           print_usage(argv[0]);
           return 1;
       }
       std::string cmd = argv[1];
       if (cmd == "run") return cmd_run(argc - 1, argv + 1);
       if (cmd == "keygen") return cmd_keygen(argc - 1, argv + 1);
       if (cmd == "version" || cmd == "--version" || cmd == "-v") return cmd_version();
       if (cmd == "help" || cmd == "--help" || cmd == "-h") { print_usage(argv[0]); return 0; }
       std::cerr << "Unknown command: " << cmd << "\n";
       print_usage(argv[0]);
       return 1;
   }
   ```

   NOTE: When passing argc/argv to cmd_run/cmd_keygen, shift by 1 so the subcommand name is removed and parse_args sees the flags correctly.

   **Hex formatting helper:** Create a simple hex encoder for namespace display:
   ```cpp
   static std::string to_hex(std::span<const uint8_t> bytes) {
       static constexpr char hex_chars[] = "0123456789abcdef";
       std::string result;
       result.reserve(bytes.size() * 2);
       for (auto b : bytes) {
           result += hex_chars[(b >> 4) & 0xF];
           result += hex_chars[b & 0xF];
       }
       return result;
   }
   ```
