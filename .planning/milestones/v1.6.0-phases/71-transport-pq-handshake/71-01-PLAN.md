    ```python
    MAX_FRAME_SIZE: int = 115343360  # 110 * 1024 * 1024 (110 MiB), matches db/net/framing.h

    def make_nonce(counter: int) -> bytes:
        """Build 12-byte AEAD nonce: [4 zero bytes][8-byte BE counter]."""
        return b"\x00\x00\x00\x00" + struct.pack(">Q", counter)

    async def send_raw(writer: asyncio.StreamWriter, data: bytes) -> None:
        """Send length-prefixed frame: [4-byte BE len(data)][data]."""
        writer.write(struct.pack(">I", len(data)) + data)
        await writer.drain()

    async def recv_raw(reader: asyncio.StreamReader) -> bytes:
        """Read length-prefixed frame. Raises ProtocolError if > MAX_FRAME_SIZE."""
        # readexactly(4) for header, unpack ">I", validate <= MAX_FRAME_SIZE
        # readexactly(length) for payload
        # Catch asyncio.IncompleteReadError -> raise chromatindb ConnectionError

    async def send_encrypted(
        writer: asyncio.StreamWriter, plaintext: bytes,
        key: bytes, counter: int,
    ) -> int:
        """Encrypt plaintext with AEAD, send as raw frame, return counter+1.
        Uses empty associated data (b"")."""
        nonce = make_nonce(counter)
        ciphertext = aead_encrypt(plaintext, b"", nonce, key)
        await send_raw(writer, ciphertext)
        return counter + 1

    async def recv_encrypted(
        reader: asyncio.StreamReader, key: bytes, counter: int,
    ) -> tuple[bytes, int]:
        """Read raw frame, decrypt with AEAD, return (plaintext, counter+1).
        Uses empty associated data (b""). Raises DecryptionError on failure."""
        ciphertext = await recv_raw(reader)
        nonce = make_nonce(counter)
        plaintext = aead_decrypt(ciphertext, b"", nonce, key)
        if plaintext is None:
            raise DecryptionError("AEAD decryption failed -- nonce desync or corrupted frame")
        return plaintext, counter + 1
    ```
