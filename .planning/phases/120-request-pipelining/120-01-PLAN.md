```cpp
// =============================================================================
// send_async() / recv_for() — pipelining primitives
// =============================================================================

bool Connection::send_async(MsgType type, std::span<const uint8_t> payload,
                            uint32_t request_id) {
    if (!connected_) return false;

    // Backpressure (D-06): pump recv() until we have a free slot. Each drained
    // reply lands in pending_replies_ for the next recv_for() to consume.
    // Single-sender invariant (D-09, PIPE-02) preserved by construction:
    // the pump only calls recv() (which never invokes send_encrypted), so the
    // sole writer of send_counter_ is still the caller of send().
    while (in_flight_ >= kPipelineDepth) {
        auto msg = recv();
        if (!msg) return false;
        // D-04: bounded by kPipelineDepth; insert_or_assign handles the
        // pathological "two replies for the same rid" case without growth.
        pending_replies_.insert_or_assign(msg->request_id, std::move(*msg));
    }

    if (!send(type, payload, request_id)) return false;
    ++in_flight_;
    return true;
}

std::optional<DecodedTransport> Connection::recv_for(uint32_t request_id) {
    if (!connected_) return std::nullopt;

    // Fast path: already stashed by a prior pump cycle.
    if (auto it = pending_replies_.find(request_id);
        it != pending_replies_.end()) {
        DecodedTransport msg = std::move(it->second);
        pending_replies_.erase(it);
        if (in_flight_ > 0) --in_flight_;
        return msg;
    }

    // Pump until target arrives.
    while (true) {
        auto msg = recv();
        if (!msg) return std::nullopt;

        if (msg->request_id == request_id) {
            if (in_flight_ > 0) --in_flight_;
            return msg;
        }

        // Off-target — stash for a future recv_for(). D-04: bounded.
        // If a reply arrives for a rid no caller will ever ask for (server
        // bug), it lingers until close() clears the map. Logging at debug
        // makes this observable without spamming the normal path.
        spdlog::debug("recv_for: stashing off-target reply rid={} (waiting for {})",
                      msg->request_id, request_id);
        pending_replies_.insert_or_assign(msg->request_id, std::move(*msg));
    }
}
```

**Step 1 — Create `cli/src/pipeline_pump.h` (header-only).**

The helper contains two free templates in `namespace chromatindb::cli::pipeline` (matches the project namespace used by `connection.h` and other `cli/src` headers; verified by grep above). The helper itself has zero `spdlog` dependency — observability is injected via an optional callback so the helper stays pure / single-responsibility / cheap to unit-test. The helper file has no `.cpp` counterpart.

```cpp
#pragma once

#include "cli/src/wire.h"

#include <cstddef>
#include <cstdint>
#include <optional>
#include <unordered_map>
#include <utility>

namespace chromatindb::cli::pipeline {

/// Drain one off-target reply into `pending` so a slot frees up. Used by
/// send_async's backpressure path. `Source` must be invocable as
/// `std::optional<DecodedTransport>()`. Returns false if the source is dead.
template <typename Source>
bool pump_one_for_backpressure(
    Source&& source,
    std::unordered_map<uint32_t, DecodedTransport>& pending) {
    auto msg = source();
    if (!msg) return false;
    pending.insert_or_assign(msg->request_id, std::move(*msg));
    return true;
}

/// No-op observer used as the default for pump_recv_for's `on_off_target`
/// parameter. Inlines away when the caller doesn't supply an observer.
struct NoopOffTargetObserver {
    void operator()(uint32_t /*off_rid*/) const noexcept {}
};

/// Pump replies until the one matching `target_rid` is found. Off-target
/// replies are stashed in `pending`. If `pending` already contains the
/// target, `source` is NEVER called. Decrements `in_flight` by one on
/// successful return. Returns std::nullopt on source error.
///
/// `on_off_target` is invoked once per off-target reply with that reply's
/// rid. Defaults to a no-op so the helper has no logging dependency; the
/// `Connection::recv_for` wrapper passes a lambda that does spdlog::debug.
/// This keeps the helper pure (single-responsibility, zero spdlog dep) and
/// keeps the Phase 120 D-04 / must-haves truth #5 observability promise
/// (the unknown-rid debug log) intact at the wrapper site.
template <typename Source, typename OnOffTarget = NoopOffTargetObserver>
std::optional<DecodedTransport> pump_recv_for(
    uint32_t target_rid,
    Source&& source,
    std::unordered_map<uint32_t, DecodedTransport>& pending,
    size_t& in_flight,
    OnOffTarget on_off_target = OnOffTarget{}) {
    if (auto it = pending.find(target_rid); it != pending.end()) {
        DecodedTransport msg = std::move(it->second);
        pending.erase(it);
        if (in_flight > 0) --in_flight;
        return msg;
    }
    while (true) {
        auto msg = source();
        if (!msg) return std::nullopt;
        if (msg->request_id == target_rid) {
            if (in_flight > 0) --in_flight;
            return msg;
        }
        on_off_target(msg->request_id);
        pending.insert_or_assign(msg->request_id, std::move(*msg));
    }
}

} // namespace chromatindb::cli::pipeline
```

```cpp
bool Connection::send_async(MsgType type, std::span<const uint8_t> payload,
                            uint32_t request_id) {
```

```cpp
// Source fixture: a FIFO queue of replies. Each call dequeues one.
struct ScriptedSource {
    std::deque<DecodedTransport> queue;
    bool dead = false;
    std::optional<DecodedTransport> operator()() {
        if (dead) return std::nullopt;
        if (queue.empty()) {
            // Test bug: pump asked for more than scripted.
            dead = true;
            return std::nullopt;
        }
        auto m = std::move(queue.front());
        queue.pop_front();
        return m;
    }
};

} // namespace
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| node → cdb client | The remote chromatindb node is the source of replies the cdb client decrypts and matches by rid. A buggy or malicious node could send replies with unexpected rids, duplicate rids, or never reply at all. AEAD already integrity-protects each frame; rid mismatch is the residual surface this phase introduces. |
| recv() pump → pending_replies_ map | Off-target replies are written into an in-memory map keyed by attacker-controlled rid (the rid is in the AEAD-protected payload, but the server picks it). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-120-01 | Tampering / Spoofing | recv_for + pending_replies_ | mitigate | rid mismatch cannot exfiltrate or corrupt — each reply is independently AEAD-decrypted in `recv()` before it ever reaches the pump (`connection.cpp` recv_encrypted at line 305 already verifies). The pump only routes already-authenticated replies. Document the AEAD precondition in the recv_for doc-comment. |
| T-120-02 | Denial of Service (memory growth) | pending_replies_ map | mitigate | Map is bounded by `Connection::kPipelineDepth` (8) by construction: `send_async` only allows `in_flight_ <= kPipelineDepth` outstanding requests, so at most 8 replies can be in flight, so `pending_replies_.size() <= kPipelineDepth` at all times. `insert_or_assign` handles duplicate-rid replies without growth (covered by Task 2 test "duplicate rid reply overwrites without growth"). On Connection destruction (close() in cli/src/connection.cpp:711-735), the map is cleared as part of the existing per-session reset. **Verified by acceptance criterion**: the [pipeline] test "send-8-receive-out-of-order correlation" asserts `pending.empty()` at end. |
| T-120-03 | Denial of Service (hang) | send_async backpressure pump | accept | A malicious server that withholds replies can make `send_async` block forever pumping `recv()` for a slot. Existing connection-level read timeouts on the asio sockets bound this (per project memory backlog 999.8 — cdb has TCP connect timeout but no PQ handshake or read deadline today; tracked separately, not introduced or worsened by this phase). Documented as an accepted residual risk; revisit when 999.8 lands a per-read deadline that will naturally cover this case too. |
| T-120-04 | Information disclosure | pending_replies_ values | accept | Stashed replies contain the same plaintext the caller would have received via recv() directly. The map lives in the same address space as the rest of the session keys; no new disclosure surface. close() zeroes session keys and clears the map next to that operation. |
| T-120-05 | Repudiation / Out-of-order delivery | Per-item completion order in cmd::get / cmd::put | accept | D-08 explicitly accepts arrival-order delivery. Phase-02 documents this in user-facing output. Not a security issue, recorded here for traceability. |
| T-120-06 | Tampering (AEAD nonce desync) | send_counter_ / recv_counter_ | mitigate | PIPE-02 / D-09: send_async only ever calls `send(...)` from the caller's thread (single-sender), and recv_for / the backpressure pump only call `recv()` (single-reader). PATTERNS.md pre-flight correction #1 confirms cli is strictly single-threaded synchronous, so no nonce desync vector is introduced. `grep` acceptance criterion in Task 1 verifies recv_for has zero `send` calls. |
| T-120-07 | Elevation of privilege | n/a | n/a | This phase touches only client-side request routing. No new authn/authz surface, no new wire fields, no new ACL paths. |
