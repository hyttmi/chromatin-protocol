    #include "relay/core/session.h"
    #include "relay/ws/ws_frame.h"

    #include <asio/ssl.hpp>
    #include <memory>
    #include <variant>

    namespace chromatindb::relay::ws {
    class SessionManager;  // Forward declare

    class WsSession : public std::enable_shared_from_this<WsSession> {
    public:
        // TLS stream type
        using TlsStream = asio::ssl::stream<asio::ip::tcp::socket>;
        // Dual-mode stream: plain TCP or TLS
        using Stream = std::variant<asio::ip::tcp::socket, TlsStream>;

        // Factory: create session after successful WS upgrade
        static std::shared_ptr<WsSession> create(
            Stream stream,
            SessionManager& manager,
            asio::any_io_executor executor,
            size_t max_send_queue);

        // Start the session lifecycle (read loop + drain + ping)
        void start(uint64_t session_id);

        // Initiate graceful close with status code + reason
        void close(uint16_t code = CLOSE_NORMAL, std::string_view reason = "");

        uint64_t id() const { return session_id_; }

        // Called by Session drain coroutine via WriteCallback
        asio::awaitable<bool> write_frame(std::span<const uint8_t> data);

    private:
        WsSession(Stream stream, SessionManager& manager,
                  asio::any_io_executor executor, size_t max_send_queue);

        // Main read loop: reads raw bytes, parses frames, handles control frames
        asio::awaitable<void> read_loop();

        // Keepalive: ping every 30s, disconnect if no pong in 60s (per D-15)
        asio::awaitable<void> ping_loop();

        // Handle a complete assembled message (text or binary)
        void on_message(uint8_t opcode, std::vector<uint8_t> data);

        // Handle control frames inline
        asio::awaitable<void> handle_control(const AssembledMessage& msg);

        // Low-level async read/write that works for both plain and TLS streams
        asio::awaitable<size_t> async_read(asio::mutable_buffer buf);
        asio::awaitable<size_t> async_write(asio::const_buffer buf);

        // Close the underlying socket
        void shutdown_socket();

        Stream stream_;
        core::Session session_;         // Send queue (from core/)
        SessionManager& manager_;
        FragmentAssembler assembler_;
        uint64_t session_id_ = 0;
        bool closing_ = false;

        // Keepalive state
        std::chrono::steady_clock::time_point last_pong_;

        // Read buffer (persistent across reads)
        static constexpr size_t READ_BUF_SIZE = 8192;
        std::array<uint8_t, READ_BUF_SIZE> read_buf_;
        std::vector<uint8_t> pending_data_;  // Accumulated unparsed bytes
    };
    }

    **relay/ws/ws_session.cpp** -- Implementation details:

    Constructor: Initialize `session_(executor, max_send_queue)`, store stream via std::move, set `last_pong_ = now()`. Call `session_.set_write_callback([self = shared_from_this()](std::span<const uint8_t> data) -> asio::awaitable<bool> { return self->write_frame(data); });` NOTE: set_write_callback must be called in start(), not constructor, because shared_from_this() is not available in constructor.

    `start(session_id)`:
    - Store session_id_.
    - Set write callback on session_ using shared_from_this().
    - co_spawn three coroutines on the executor: read_loop(), session_.drain_send_queue(), ping_loop(). All use asio::detached.

    `async_read(buf)` / `async_write(buf)`:
    - Use std::visit on stream_:
      - For tcp::socket: `co_await socket.async_read_some(buf, asio::use_awaitable)`
      - For TlsStream: `co_await tls.async_read_some(buf, asio::use_awaitable)`
    - Return bytes transferred. On error, return 0 or propagate.

    `write_frame(data)`:
    - Encode as text frame using `encode_frame(OPCODE_TEXT, data)`. (Phase 102+ will distinguish text vs binary based on message type.)
    - Call `co_await async_write(asio::buffer(frame))`.
    - Return true on success, false on write error (triggers session close).

    `read_loop()`:
    - Infinite loop: `auto n = co_await async_read(asio::buffer(read_buf_))`. On error or n==0, break.
    - Append read_buf_[0..n) to pending_data_.
    - Inner parse loop: try to parse a frame header from pending_data_. If nullopt (insufficient bytes), break inner loop and read more.
    - If header parsed and enough payload bytes available in pending_data_:
      - Validate: if client frame has MASK=0, send Close(1002 "Client frames must be masked" per D-10), break.
      - Extract payload bytes from pending_data_, apply_mask if masked.
      - Feed into assembler_.feed(opcode, fin, payload).
      - Switch on AssemblyResult:
        - COMPLETE: call on_message(msg.opcode, std::move(msg.data)).
        - CONTROL: co_await handle_control(msg).
        - ERROR: send Close(1002, msg.error_reason), break.
        - INCOMPLETE: continue reading.
      - Erase consumed bytes from pending_data_ (header_size + payload_length).
    - After loop exits: remove session from manager, close session.

    `handle_control(msg)`:
    - OPCODE_PING: respond with Pong (same payload). `auto pong = encode_frame(OPCODE_PONG, msg.data); co_await async_write(asio::buffer(pong));`
    - OPCODE_PONG: update `last_pong_ = now()`.
    - OPCODE_CLOSE: echo close frame with same status code (per D-16). `auto close = encode_close_frame(msg.close_code); co_await async_write(asio::buffer(close));` Then shutdown.

    `ping_loop()` (per D-15):
    - Loop every 30 seconds: `asio::steady_timer timer(co_await asio::this_coro::executor); timer.expires_after(30s); co_await timer.async_wait(use_nothrow);`
    - If closing_, break.
    - Check if `now() - last_pong_ > 60s`. If so, log warning, close(1002, "Pong timeout"), break.
    - Send Ping frame: `auto ping = encode_frame(OPCODE_PING, {}); co_await async_write(asio::buffer(ping));`

    `on_message(opcode, data)`:
    - Phase 101: log the message at debug level ("received {} frame, {} bytes", opcode==1?"text":"binary", data.size()). No further processing until Phase 102 adds JSON parsing.

    `close(code, reason)`:
    - If already closing_, return.
    - Set closing_ = true.
    - Send close frame: `session_.enqueue(encode_close_frame(code, reason))`.
    - Set a 5-second timer (per D-16). After timeout, shutdown_socket().
    - session_.close() to drain the queue.
    - manager_.remove_session(session_id_).

    `shutdown_socket()`:
    - std::visit on stream_: for tcp::socket call shutdown(both) + close(). For TlsStream, call async_shutdown then close underlying socket.

    **relay/ws/ws_acceptor.cpp** -- Implementation:

    Constructor: Create tcp::acceptor on bind_address:bind_port. Set SO_REUSEADDR. Open, bind, listen. Support both IPv4 and IPv6 (per D-22): try to resolve bind_address, use the first endpoint family.

    `init_tls(cert, key)`:
    - Create `auto ctx = std::make_shared<asio::ssl::context>(asio::ssl::context::tlsv13_server);` (per D-18, D-20)
    - `ctx->use_certificate_chain_file(cert);`
    - `ctx->use_private_key_file(key, asio::ssl::context::pem);`
    - Lock mutex, store ctx as tls_ctx_. Return true.
    - On exception: log error, return false.

    `reload_tls(cert, key)` (per D-19):
    - Same as init_tls but logs "TLS context reloaded" on success, "TLS reload failed: {}" on failure.
    - On failure: do NOT replace tls_ctx_ (keep old context). Return false.

    `is_tls_enabled()`: Lock mutex, return `tls_ctx_ != nullptr`.

    `accept_loop()`:
    - While !stopping_:
      - `auto [ec, socket] = co_await acceptor_.async_accept(asio::as_tuple(asio::use_awaitable));`
      - If ec or stopping_, co_return.
      - If `manager_.count() >= MAX_CONNECTIONS` (per D-17): log warn "connection rejected: max connections (1024) reached", close socket, continue.
      - `asio::co_spawn(ioc_, handle_new_connection(std::move(socket)), asio::detached);`

    `handle_new_connection(socket)`:
    - Start a 5-second timer (HANDSHAKE_TIMEOUT per D-13) that closes the socket on expiry.
    - If TLS enabled:
      - Lock mutex, copy tls_ctx_ shared_ptr. Unlock.
      - Create `asio::ssl::stream<tcp::socket> tls_stream(std::move(socket), *tls_ctx)`. The ssl::stream holds a reference to ssl::context, and the shared_ptr keeps it alive.
      - `co_await tls_stream.async_handshake(asio::ssl::stream_base::server, asio::use_awaitable);`
      - On error: log at debug "TLS handshake failed: {}", return. (Per D-06: no session created.)
      - Cancel handshake timer.
      - Perform WS upgrade: `co_await do_ws_upgrade(tls_stream)`.
      - If upgrade succeeds: Create WsSession with TlsStream variant, register in manager, start().
    - If plain WS (per D-03: warn on startup, not per-connection):
      - Perform WS upgrade directly: `co_await do_ws_upgrade(socket)`.
      - If upgrade succeeds: Create WsSession with tcp::socket variant, register in manager, start().
    - Cancel handshake timer if still running.

    `do_ws_upgrade(stream)` (template):
    - Read up to 4096 bytes looking for HTTP upgrade request (read until "\r\n\r\n" found in buffer).
    - Call `parse_upgrade_request(raw)`.
    - If `!result.is_upgrade`: write `build_426_response()` (per D-12), return false.
    - If `result.is_upgrade` but `!result.request`: log warn with result.error, close, return false.
    - Compute accept key: `compute_accept_key(result.request->websocket_key)`.
    - Write `build_upgrade_response(accept_key)`.
    - Return true.

    `stop()`: Set stopping_ = true. Close acceptor. (Wakes async_accept with error.)
