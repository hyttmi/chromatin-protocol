**The challenge:** Currently, HttpConnection::handle() reads the full body BEFORE dispatching to the router. For streaming uploads, we need to start forwarding chunks to UDS while still reading from the HTTP socket. This requires the handler to have access to the HttpConnection to call read_body_chunked().

**A. Modify dispatch model for streaming routes:**

The current flow is:
1. handle() reads full body
2. handle() calls router_.dispatch_async(req, body, token_store_)
3. Router calls handler(req, body, session) -> HttpResponse
4. handle() serializes and sends response

For streaming writes, the flow must be:
1. handle() reads headers, checks Content-Length
2. If POST /blob AND Content-Length >= STREAMING_THRESHOLD:
   - handle() calls a streaming handler variant that receives the HttpConnection reference
   - Streaming handler reads body chunks from connection, forwards to UDS
   - Streaming handler awaits WriteAck and sends response directly
   - handle() does NOT read body or send response (streaming handler did both)
3. Else: existing full-buffer path

asio::awaitable<HttpResponse> DataHandlers::handle_blob_write_streaming(
    const HttpRequest& req, HttpConnection& conn, HttpSessionState* session) {

    // 1. Size limit check
    if (max_blob_size_ > 0 && req.content_length > max_blob_size_) {
        // Must drain body to keep connection clean
        std::vector<uint8_t> discard;
        co_await conn.read_body(req.content_length, discard);
        co_return HttpResponse::error(413, "payload_too_large",
            "body size " + std::to_string(req.content_length) +
            " exceeds limit " + std::to_string(max_blob_size_));
    }

    // 2. Register request with RequestRouter
    uint32_t relay_rid = router_.register_request(session->session_id, 0, 8);

    // 3. Accumulate entire body but in chunks, building the full payload
    // for the UDS chunked send.
    //
    // NOTE: For the streaming UPLOAD path, we still need the full payload
    // because the node's framing currently expects a complete Data(8)
    // FlatBuffer blob in the reassembled chunked message. The benefit
    // of chunked UDS send is that drain_send_queue sends chunks one at
    // a time (each AEAD-encrypted independently) rather than one massive
    // AEAD encrypt of the full payload.
    //
    // True streaming (where the relay never holds the full body) requires
    // node-side storage streaming, which is deferred per research Open
    // Question 1.
    std::vector<uint8_t> full_body;
    full_body.reserve(req.content_length);

    bool read_ok = co_await conn.read_body_chunked(
        req.content_length, core::CHUNK_SIZE,
        [&full_body](std::span<const uint8_t> chunk) -> asio::awaitable<bool> {
            full_body.insert(full_body.end(), chunk.begin(), chunk.end());
            co_return true;
        });

    if (!read_ok) {
        promises_.remove(relay_rid);
        co_return HttpResponse::error(400, "read_error", "failed to read request body");
    }

    // 4. Encode via TransportCodec (creates FlatBuffer wrapper)
    auto transport_msg = wire::TransportCodec::encode(
        chromatindb::wire::TransportMsgType_Data,
        std::span<const uint8_t>(full_body), relay_rid);

    // 5. Send via UDS — use chunked path since payload >= STREAMING_THRESHOLD.
    // The TransportCodec-encoded message is the payload that gets chunked.
    bool sent = uds_mux_.send_chunked_msg(
        8, relay_rid, std::move(transport_msg));
    if (!sent) {
        co_return HttpResponse::error(502, "uds_down", "node connection unavailable");
    }

    // Wait: Actually, send_chunked_msg sends via the chunked sub-frame protocol,
    // where the chunked header replaces TransportCodec. We should NOT double-encode.
    // The chunked header carries type + request_id. The payload should be the RAW
    // blob data, not a TransportCodec-wrapped message.
    //
    // BUT: the node's recv_chunked() reassembles and then calls message_cb_
    // which expects type + payload (same as TransportCodec::decode would return).
    // The chunked header provides type and request_id. The payload is what goes
    // into message_cb_'s payload parameter.
    //
    // For Data(8), the node's on_peer_message handler expects the payload to be
    // the raw FlatBuffer blob (same as what TransportCodec::encode puts in the
    // payload field). So we should send the RAW body (the FlatBuffer blob),
    // not the TransportCodec-wrapped version.

    // Corrected: send raw body via chunked UDS (no TransportCodec wrapping)
    // The chunked header carries type=8 and request_id=relay_rid
    bool sent_ok = uds_mux_.send_chunked_msg(
        8, relay_rid, std::move(full_body));
    if (!sent_ok) {
        promises_.remove(relay_rid);
        co_return HttpResponse::error(502, "uds_down", "node connection unavailable");
    }

    // 6. Create promise and await WriteAck response
    auto promise = promises_.create_promise(relay_rid, ioc_.get_executor());
    auto timeout_sec = request_timeout_;
    if (timeout_sec == 0) timeout_sec = 10;
    auto result = co_await promise->wait(std::chrono::seconds(timeout_sec));
    promises_.remove(relay_rid);

    if (!result) {
        co_return HttpResponse::error(504, "timeout", "node response timeout");
    }

    // 7. Expect WriteAck (type 30)
    if (result->type != 30) {
        co_return HttpResponse::error(502, "unexpected_response",
            "unexpected response type from node");
    }

    // 8. Source exclusion
    if (result->payload.size() >= 32) {
        core::BlobHash32 blob_hash{};
        std::memcpy(blob_hash.data(), result->payload.data(), 32);
        write_tracker_.record(blob_hash, session->session_id);
    }

    // 9. Translate WriteAck -> JSON
    auto json_opt = co_await util::offload_if_large(pool_, ioc_,
        result->payload.size(),
        [&] { return translate::binary_to_json(result->type,
            std::span<const uint8_t>(result->payload)); });
    if (!json_opt) {
        co_return HttpResponse::error(502, "decode_error", "failed to decode WriteAck");
    }

    co_return HttpResponse::json(200, *json_opt);
}
```

**Streaming blob read is more complex than write** because the response payload comes from UDS asynchronously and must be streamed to the HTTP client via chunked transfer encoding.

The key challenge: the current ResponsePromise delivers the ENTIRE payload at once. For streaming reads, we need the read_loop() to deliver chunks incrementally to the handler coroutine.

**A. Add StreamingResponsePromise to response_promise.h:**

For streaming downloads, instead of collecting the full payload in ResponsePromise, we need a mechanism where read_loop() pushes chunks and the handler coroutine consumes them.

But wait — with the current architecture (Plan 02), the relay's read_loop() already reassembles chunked responses via recv_chunked_reassemble() into a complete payload before calling route_response(). For true streaming, we would need read_loop() to NOT reassemble but instead push chunks to a ChunkQueue.

**Revised approach for streaming downloads:**

Option A: Have read_loop() deliver the full reassembled payload via ResponsePromise (current after Plan 02). The handler then streams it to the HTTP client chunk by chunk from memory. This still buffers the full blob in memory but avoids the AEAD-all-at-once bottleneck (each chunk was individually AEAD'd on the thread pool).

Option B: Modify read_loop() to detect when a chunked response should be streamed to HTTP and push chunks to a ChunkQueue instead of reassembling. The handler coroutine consumes from the queue and writes chunked-TE to HTTP.

Option B is the correct streaming approach but requires changing how read_loop() dispatches chunked responses for specific request_ids. Let's implement Option B.

/// Streaming response for large blob reads.
/// read_loop() pushes chunks; handler coroutine pops and streams to HTTP.
struct StreamingResponsePromise {
    explicit StreamingResponsePromise(asio::any_io_executor executor)
        : header_timer_(executor), queue(executor) {}

    /// Header info (type, status byte, etc.) delivered before chunks.
    struct HeaderInfo {
        uint8_t type;
        uint32_t request_id;
        uint64_t total_size;
        std::vector<uint8_t> extra_metadata;  // e.g., status byte for ReadResponse
    };

    /// Set header info (called by read_loop when chunked header received).
    void set_header(HeaderInfo info) {
        header_ = std::move(info);
        header_ready_ = true;
        header_timer_.cancel();
    }

    /// Wait for header info. Returns nullopt on timeout.
    template<typename Duration>
    asio::awaitable<std::optional<HeaderInfo>> wait_header(Duration timeout) {
        if (header_ready_) co_return std::move(header_);
        header_timer_.expires_after(timeout);
        auto [ec] = co_await header_timer_.async_wait(asio::as_tuple(asio::use_awaitable));
        if (header_ready_) co_return std::move(header_);
        co_return std::nullopt;
    }

    /// Chunk queue for incremental delivery.
    core::ChunkQueue queue;

private:
    asio::steady_timer header_timer_;
    std::optional<HeaderInfo> header_;
    bool header_ready_ = false;
};
```

DataHandlers::handle_blob_read_streaming(
    const HttpRequest& req, HttpConnection& conn, HttpSessionState* session) {

    // 1. Extract namespace and hash from path (same as non-streaming)
    auto params = extract_blob_path_params(req.path, "/blob/");
    if (!params) {
        co_return HttpResponse::error(400, "invalid_path",
            "expected /blob/{namespace_hex_64}/{hash_hex_64}");
    }

    // 2. Build ReadRequest payload: [namespace:32][hash:32] = 64 bytes
    std::vector<uint8_t> payload(64);
    std::memcpy(payload.data(), params->namespace_bytes.data(), 32);
    std::memcpy(payload.data() + 32, params->hash_bytes.data(), 32);

    // 3. Register request
    uint32_t relay_rid = router_.register_request(session->session_id, 0, 31);

    // 4. Create STREAMING promise (not regular promise)
    auto streaming_promise = promises_.create_streaming_promise(
        relay_rid, ioc_.get_executor());

    // 5. Encode ReadRequest and send via UDS (small message, use regular send)
    auto transport_msg = wire::TransportCodec::encode(
        chromatindb::wire::TransportMsgType_ReadRequest,
        std::span<const uint8_t>(payload), relay_rid);
    bool sent = uds_mux_.send(std::move(transport_msg));
    if (!sent) {
        promises_.remove_streaming(relay_rid);
        co_return HttpResponse::error(502, "uds_down", "node connection unavailable");
    }

    // 6. Wait for header (contains status byte in extra_metadata per Pitfall 4)
    auto timeout_sec = request_timeout_;
    if (timeout_sec == 0) timeout_sec = 10;
    auto header = co_await streaming_promise->wait_header(
        std::chrono::seconds(timeout_sec));

    if (!header) {
        promises_.remove_streaming(relay_rid);
        co_return HttpResponse::error(504, "timeout", "node response timeout");
    }

    // 7. Check type = ReadResponse (32)
    if (header->type != 32) {
        // Drain any pending chunks
        streaming_promise->queue.close();
        promises_.remove_streaming(relay_rid);
        co_return HttpResponse::error(502, "unexpected_response",
            "unexpected response type from node");
    }

    // 8. Check status byte (Pitfall 4: must check BEFORE committing to 200)
    // ReadResponse: status byte is the first byte of the reassembled payload.
    // For chunked mode, the status byte should be in extra_metadata of the chunked header.
    // If extra_metadata is empty, status byte will be at the start of the first data chunk.
    uint8_t status = 0;
    if (!header->extra_metadata.empty()) {
        status = header->extra_metadata[0];
    } else {
        // Fallback: read first chunk and check
        auto first_chunk = co_await streaming_promise->queue.pop();
        if (!first_chunk || first_chunk->empty()) {
            promises_.remove_streaming(relay_rid);
            co_return HttpResponse::not_found();
        }
        status = (*first_chunk)[0];
        // If found, put remaining bytes back... this is messy.
        // Better: require extra_metadata to carry status byte (Plan 01 header format supports it)
    }

    if (status != 0x01) {
        // Not found — drain remaining chunks and return 404
        streaming_promise->queue.close();
        promises_.remove_streaming(relay_rid);
        co_return HttpResponse::not_found();
    }

    // 9. Start HTTP chunked transfer encoding (commit to 200)
    bool te_ok = co_await conn.write_chunked_te_start(200, "application/octet-stream");
    if (!te_ok) {
        streaming_promise->queue.close();
        promises_.remove_streaming(relay_rid);
        co_return std::nullopt;  // Connection dead, no response to send
    }

    // 10. Stream chunks from UDS -> HTTP
    while (true) {
        auto chunk = co_await streaming_promise->queue.pop();
        if (!chunk) break;  // Queue closed = all chunks delivered
        if (chunk->empty()) break;

        bool write_ok = co_await conn.write_chunked_te_chunk(*chunk);
        if (!write_ok) {
            // HTTP client disconnected mid-stream (Pitfall 2)
            streaming_promise->queue.close();
            break;
        }
    }

    // 11. End chunked TE
    co_await conn.write_chunked_te_end();

    // 12. Cleanup
    promises_.remove_streaming(relay_rid);

    // Return nullopt to signal handle() that response was already written
    co_return std::nullopt;
}

**E. Add streaming read dispatch to HttpConnection::handle():**

For GET /blob/ reads, we don't know the response size until the UDS response arrives. The streaming/non-streaming decision happens AFTER we send ReadRequest and see what comes back. Two approaches:

Approach 1: Always use streaming promise for GET /blob/, let the handler decide based on response size.
Approach 2: Use regular promise, and if the response is large, re-stream from the reassembled payload.

Approach 2 is simpler and avoids the streaming promise complexity for reads. Since read_loop() already reassembles chunked responses (Plan 02), the handler gets the full payload via regular ResponsePromise. Then the handler streams it to HTTP chunk by chunk from memory.

This still buffers the full blob, but the UDS AEAD was per-chunk (off the event loop), and the HTTP write is chunked. The main benefit is per-chunk AEAD offloading and HTTP chunked-TE (no Content-Length needed). True zero-copy streaming requires much more plumbing.

DataHandlers::handle_blob_read_streaming(
    const HttpRequest& req, HttpConnection& conn, HttpSessionState* session) {

    auto params = extract_blob_path_params(req.path, "/blob/");
    if (!params) {
        co_return HttpResponse::error(400, "invalid_path", "...");
    }

    std::memcpy(payload.data(), params->namespace_bytes.data(), 32);
    std::memcpy(payload.data() + 32, params->hash_bytes.data(), 32);

    uint32_t relay_rid = router_.register_request(session->session_id, 0, 31);
    auto transport_msg = wire::TransportCodec::encode(
        chromatindb::wire::TransportMsgType_ReadRequest,
        std::span<const uint8_t>(payload), relay_rid);

    // Use regular send_and_await (reassembled response via Plan 02 chunked recv)
    auto result = co_await send_and_await(std::move(transport_msg), relay_rid);
    if (!result) co_return HttpResponse::error(504, "timeout", "node response timeout");

    if (result->type != 32) {
        co_return HttpResponse::error(502, "unexpected_response", "...");
    }

    // Check status byte
    if (result->payload.empty() || result->payload[0] != 0x01) {
        co_return HttpResponse::not_found();
    }

    // Payload data starts after status byte
    std::span<const uint8_t> blob_data(
        result->payload.data() + 1, result->payload.size() - 1);

    // For large blobs: stream via chunked-TE
    if (blob_data.size() >= core::STREAMING_THRESHOLD) {
        // Start chunked transfer encoding
        bool te_ok = co_await conn.write_chunked_te_start(200, "application/octet-stream");
        if (!te_ok) co_return std::nullopt;

        // Write data in CHUNK_SIZE chunks
        size_t offset = 0;
        while (offset < blob_data.size()) {
            size_t chunk_len = std::min(core::CHUNK_SIZE, blob_data.size() - offset);
            auto chunk_span = blob_data.subspan(offset, chunk_len);
            bool ok = co_await conn.write_chunked_te_chunk(chunk_span);
            if (!ok) co_return std::nullopt;  // Client disconnected
            offset += chunk_len;
        }

        co_await conn.write_chunked_te_end();
        co_return std::nullopt;  // Response already written
    }

    // Small blob: regular response
    std::vector<uint8_t> data(blob_data.begin(), blob_data.end());
    co_return HttpResponse::binary(200, std::move(data));
}
