```cpp
int cmd_add_peer(int argc, char* argv[]) {
    std::string data_dir = "./data";
    std::string config_path;
    std::string address;

    for (int i = 1; i < argc; ++i) {
        std::string arg = argv[i];
        if (arg == "--data-dir" && i + 1 < argc) {
            data_dir = argv[++i];
        } else if (arg == "--config" && i + 1 < argc) {
            config_path = argv[++i];
        } else if (address.empty() && arg[0] != '-') {
            address = arg;
        }
    }

    if (address.empty()) {
        std::cerr << "Usage: chromatindb add-peer <host:port> [--config <path>] [--data-dir <path>]\n";
        return 1;
    }

    // Resolve config path
    if (config_path.empty()) {
        config_path = (std::filesystem::path(data_dir) / "config.json").string();
    }

    try {
        // Read existing config as raw JSON (preserves all fields)
        nlohmann::json j;
        {
            std::ifstream f(config_path);
            if (f.is_open()) {
                j = nlohmann::json::parse(f);
            }
        }

        // Ensure bootstrap_peers is an array
        if (!j.contains("bootstrap_peers") || !j["bootstrap_peers"].is_array()) {
            j["bootstrap_peers"] = nlohmann::json::array();
        }

        // Check for duplicates
        for (const auto& peer : j["bootstrap_peers"]) {
            if (peer.get<std::string>() == address) {
                std::cerr << "Peer " << address << " already in bootstrap_peers" << std::endl;
                return 1;
            }
        }

        // Add the peer
        j["bootstrap_peers"].push_back(address);

        // Write back with indentation
        {
            std::ofstream f(config_path);
            if (!f.is_open()) {
                std::cerr << "Error: cannot write " << config_path << std::endl;
                return 1;
            }
            f << j.dump(4) << std::endl;
        }

        std::cout << "Added peer " << address << " to " << config_path << std::endl;

        // Send SIGHUP to running node if pidfile exists (D-10)
        auto pidfile = std::filesystem::path(data_dir) / "chromatindb.pid";
        if (std::filesystem::exists(pidfile)) {
            std::ifstream pf(pidfile);
            pid_t pid;
            if (pf >> pid && kill(pid, 0) == 0) {
                kill(pid, SIGHUP);
                std::cout << "Sent SIGHUP to node (PID " << pid << ")" << std::endl;
            }
        }

        return 0;
    } catch (const std::exception& e) {
        std::cerr << "Error: " << e.what() << std::endl;
        return 1;
    }
```

```cpp
int cmd_remove_peer(int argc, char* argv[]) {
    std::string data_dir = "./data";
    std::string config_path;
    std::string address;

    for (int i = 1; i < argc; ++i) {
        std::string arg = argv[i];
        if (arg == "--data-dir" && i + 1 < argc) {
            data_dir = argv[++i];
        } else if (arg == "--config" && i + 1 < argc) {
            config_path = argv[++i];
        } else if (address.empty() && arg[0] != '-') {
            address = arg;
        }
    }

    if (address.empty()) {
        std::cerr << "Usage: chromatindb remove-peer <host:port> [--config <path>] [--data-dir <path>]\n";
        return 1;
    }

    // Resolve config path
    if (config_path.empty()) {
        config_path = (std::filesystem::path(data_dir) / "config.json").string();
    }

    try {
        // Read existing config as raw JSON
        nlohmann::json j;
        {
            std::ifstream f(config_path);
            if (!f.is_open()) {
                std::cerr << "Error: cannot read " << config_path << std::endl;
                return 1;
            }
            j = nlohmann::json::parse(f);
        }

        if (!j.contains("bootstrap_peers") || !j["bootstrap_peers"].is_array()) {
            std::cerr << "No bootstrap_peers array in " << config_path << std::endl;
            return 1;
        }

        // Find and remove the peer
        auto& peers = j["bootstrap_peers"];
        auto it = std::find(peers.begin(), peers.end(), address);
        if (it == peers.end()) {
            std::cerr << "Peer " << address << " not found in bootstrap_peers" << std::endl;
            return 1;
        }
        peers.erase(it);

        // Write back
        {
            std::ofstream f(config_path);
            if (!f.is_open()) {
                std::cerr << "Error: cannot write " << config_path << std::endl;
                return 1;
            }
            f << j.dump(4) << std::endl;
        }

        std::cout << "Removed peer " << address << " from " << config_path << std::endl;

        // Send SIGHUP to running node if pidfile exists (D-10)
        auto pidfile = std::filesystem::path(data_dir) / "chromatindb.pid";
        if (std::filesystem::exists(pidfile)) {
            std::ifstream pf(pidfile);
            pid_t pid;
            if (pf >> pid && kill(pid, 0) == 0) {
                kill(pid, SIGHUP);
                std::cout << "Sent SIGHUP to node (PID " << pid << ")" << std::endl;
            }
        }

        return 0;
    } catch (const std::exception& e) {
        std::cerr << "Error: " << e.what() << std::endl;
        return 1;
    }
}
```

**db/main.cpp** -- Implement `cmd_list_peers()` with UDS PeerInfoRequest query and config.json fallback.

This is the most complex subcommand. It needs to:
1. Read config.json for bootstrap_peers and uds_path
2. Try to connect to the running node via UDS and query peer info
3. If UDS connection fails, fall back to showing config bootstrap peers only (all marked disconnected per D-14)
4. Merge connected peers from UDS response with disconnected bootstrap peers from config (per D-12)

**Implementation approach per D-11 through D-14:**

The node's UDS socket requires TrustedHello handshake + AEAD encryption (per Pitfall 5 in RESEARCH.md). This is ~100 lines of synchronous code. The `chromatindb` binary already links liboqs, libsodium, and Asio -- all crypto primitives are available.

However, implementing a full UDS handshake client is substantial. The pragmatic approach: use the node's existing loadable config to find the UDS path, then do a synchronous TrustedHello + AEAD exchange. The key files to reference for the handshake protocol:
- `cli/src/connection.cpp` lines 139-151 (connect_uds) and 399-508 (handshake_trusted)  
- `db/net/connection.cpp` for the TrustedHello wire format

**Implement `cmd_list_peers()`:**

```cpp
int cmd_list_peers(int argc, char* argv[]) {
    std::string data_dir = "./data";
    std::string config_path;

    for (int i = 1; i < argc; ++i) {
        std::string arg = argv[i];
        if (arg == "--data-dir" && i + 1 < argc) {
            data_dir = argv[++i];
        } else if (arg == "--config" && i + 1 < argc) {
            config_path = argv[++i];
        }
    }

    if (config_path.empty()) {
        config_path = (std::filesystem::path(data_dir) / "config.json").string();
    }

    try {
        // Load config for bootstrap_peers and uds_path
        auto config = chromatindb::config::load_config(config_path);
        auto bootstrap_set = std::set<std::string>(
            config.bootstrap_peers.begin(), config.bootstrap_peers.end());

        // Try UDS query to running node
        bool node_running = false;
        struct PeerEntry {
            std::string address;
            bool is_bootstrap;
            bool syncing;
            bool peer_is_full;
            uint64_t duration_ms;
        };
        std::vector<PeerEntry> connected_peers;
        std::set<std::string> connected_addresses;

        if (!config.uds_path.empty()) {
            // Attempt UDS connection and PeerInfoRequest
            // This requires TrustedHello handshake + AEAD -- complex
            // For now, try a minimal approach: check if the node is running via pidfile
            // If so, note that a full UDS client would be needed
            // IMPLEMENTATION: build a minimal synchronous UDS client
        }

        // The full UDS handshake is complex (~100 lines of crypto). 
        // Implement it by loading identity keys from data_dir, connecting to UDS,
        // performing TrustedHello, HKDF, AuthSignature exchange, draining 
        // SyncNamespaceAnnounce, then sending PeerInfoRequest and parsing response.
        //
        // Reference: cli/src/connection.cpp:399-508 for the exact protocol sequence.
        // Reference: db/peer/message_dispatcher.cpp:1189-1214 for response format.
        //
        // The executor MUST read cli/src/connection.cpp to understand the handshake
        // sequence before implementing. Key steps:
        // 1. Generate random 32-byte nonce
        // 2. Load signing keypair from data_dir/identity.key
        // 3. Connect to UDS path
        // 4. Build TrustedHello: [nonce:32][signing_pk:2592], wrap as type 23 message
        //    Wire format: [1B flags=0x00][4B length_be][4B type_be=23][4B rid_be=0][payload]
        // 5. Send TrustedHello, receive TrustedHello response
        // 6. HKDF-SHA256: IKM = initiator_nonce || responder_nonce
        //    Salt = initiator_pk || responder_pk (EMPTY salt per memory notes)
        //    Info = "chromatindb-trusted-v1"
        //    Derive 64 bytes: first 32 = send_key, last 32 = recv_key
        //    (initiator uses first 32 for send, last 32 for recv)
        // 7. AEAD-encrypt AuthSignature (type 3), send it, receive peer's AuthSignature
        // 8. Drain SyncNamespaceAnnounce (type 62) -- read one message, discard if type 62
        // 9. AEAD-encrypt PeerInfoRequest (type 55, empty payload), send it
        // 10. Receive and AEAD-decrypt PeerInfoResponse (type 56), parse binary format

        // FALLBACK (D-14): If UDS connection fails or uds_path empty, show config only
        if (!node_running) {
            std::cout << "Node is not running (showing config only)\n\n";
        }

        // Print header
        std::cout << std::left
                  << std::setw(30) << "ADDRESS"
                  << std::setw(12) << "STATUS"
                  << std::setw(12) << "BOOTSTRAP"
                  << std::setw(10) << "SYNCING"
                  << std::setw(10) << "FULL"
                  << "UPTIME" << std::endl;
        std::cout << std::string(84, '-') << std::endl;

        // Print connected peers from UDS response
        for (const auto& p : connected_peers) {
            connected_addresses.insert(p.address);
            auto uptime = format_duration(p.duration_ms);
            std::cout << std::left
                      << std::setw(30) << p.address
                      << std::setw(12) << "connected"
                      << std::setw(12) << (p.is_bootstrap ? "yes" : "no")
                      << std::setw(10) << (p.syncing ? "yes" : "no")
                      << std::setw(10) << (p.peer_is_full ? "yes" : "no")
                      << uptime << std::endl;
        }

        for (const auto& addr : bootstrap_set) {
```
