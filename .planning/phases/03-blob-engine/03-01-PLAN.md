1. Create `src/engine/engine.h`:
   ```cpp
   #pragma once
   #include <array>
   #include <cstdint>
   #include <optional>
   #include <span>
   #include <string>
   #include <vector>
   #include "wire/codec.h"

   namespace chromatin::storage { class Storage; struct NamespaceInfo; }

   namespace chromatin::engine {

   enum class IngestError {
       namespace_mismatch,
       invalid_signature,
       malformed_blob,
       storage_error
   };

   enum class IngestStatus { stored, duplicate };

   struct WriteAck {
       std::array<uint8_t, 32> blob_hash{};
       uint64_t seq_num = 0;
       IngestStatus status = IngestStatus::stored;
       uint32_t replication_count = 1;  // Stubbed until ACKW-02
   };

   struct IngestResult {
       bool accepted = false;
       std::optional<WriteAck> ack;
       std::optional<IngestError> error;
       std::string error_detail;

       static IngestResult success(WriteAck ack);
       static IngestResult rejection(IngestError err, std::string detail = "");
   };

   class BlobEngine {
   public:
       explicit BlobEngine(storage::Storage& store);
       IngestResult ingest(const wire::BlobData& blob);
       // Query methods in Plan 02
   private:
       storage::Storage& storage_;
   };

   } // namespace chromatin::engine
   ```

2. Create `src/engine/engine.cpp`:
   Implement the fail-fast validation pipeline per CONTEXT.md locked decisions:
   - Step 1 (structural): Check `blob.pubkey.size() == crypto::Signer::PUBLIC_KEY_SIZE` (2592). Check `blob.signature.empty()`. If either fails, return `IngestResult::rejection(IngestError::malformed_blob, detail)`. Log at warn level.
   - Step 2 (namespace): Compute `crypto::sha3_256(blob.pubkey)` and compare to `blob.namespace_id`. If mismatch, return `IngestResult::rejection(IngestError::namespace_mismatch)`. Log with truncated namespace (first 2 bytes hex).
   - Step 3 (signature): Call `wire::build_signing_input(blob.namespace_id, blob.data, blob.ttl, blob.timestamp)` then `crypto::Signer::verify(signing_input, blob.signature, blob.pubkey)`. If false, return `IngestResult::rejection(IngestError::invalid_signature)`. Log with truncated namespace.
   - Step 4 (store): Call `storage_.store_blob(blob)`. Map the result:
     - `Status::Stored` -> `IngestResult::success(WriteAck{result.blob_hash, result.seq_num, IngestStatus::stored, 1})`
     - `Status::Duplicate` -> `IngestResult::success(WriteAck{result.blob_hash, result.seq_num, IngestStatus::duplicate, 1})`
     - `Status::Error` -> `IngestResult::rejection(IngestError::storage_error, "storage write failed")`
