<objective>
Add per-connection token bucket rate limiting for Data and Delete messages. A peer exceeding the configured byte throughput rate is immediately disconnected (no backpressure). Sync traffic is excluded. Rate parameters are SIGHUP-reloadable.

Purpose: Protect open nodes from write-flooding abuse by cutting abusive peers before they can exhaust resources.
Output: Rate-limited ingest path with configurable thresholds, metrics counter, and SIGHUP reload support.
</objective>

    ```cpp
    bool try_consume_tokens(PeerInfo& peer, uint64_t bytes,
                            uint64_t rate_bytes_per_sec, uint64_t burst_bytes) {
        auto now_ms = static_cast<uint64_t>(
            std::chrono::duration_cast<std::chrono::milliseconds>(
                std::chrono::steady_clock::now().time_since_epoch()).count());
        uint64_t elapsed_ms = now_ms - peer.bucket_last_refill;
        // Cap elapsed to prevent intermediate overflow
        uint64_t max_elapsed = (burst_bytes * 1000) / rate_bytes_per_sec + 1;
        elapsed_ms = std::min(elapsed_ms, max_elapsed);
        uint64_t refill = (rate_bytes_per_sec * elapsed_ms) / 1000;
        peer.bucket_tokens = std::min(peer.bucket_tokens + refill, burst_bytes);
        peer.bucket_last_refill = now_ms;
        if (bytes > peer.bucket_tokens) {
            return false;
        }
        peer.bucket_tokens -= bytes;
        return true;
    }
    ```
